  lcd2.clear();

  int num = n + 1;

    if (leds[i] == CRGB::White) {
      checkWhite(top - 8, num);
    }
  }
  Serial.println("n: " + n);
  return n;
}

      FastLED.show();
    }
    Serial.println("cyan Button pressed");
  }
  if (lastKey == 'a') {
    if (leds[NUM_LEDS - 7] == CRGB::Yellow || leds[NUM_LEDS - 10] == CRGB::Yellow || leds[NUM_LEDS - 23] == CRGB::Yellow) {
      leds[NUM_LEDS - 7] = CRGB::Black;
      leds[NUM_LEDS - 10] = CRGB::Black;
      leds[NUM_LEDS - 23] = CRGB::Black;
      score++;


      FastLED.show();
    }
    Serial.println("yellow Button pressed");
  }






  FastLED.show();


  delay(speed / 2);










  for (int i = 0; i < 8; i++) {
    leds[i] = CRGB::Black;
  }
  for (int i = 255; i >= 248; i--) {
    leds[i] = CRGB::Black;
  }




  if (speed > 150) {
    speed -= 2;
  }
  Serial.print("Speed: ");
  Serial.println(speed);




  lastKey = 'd';
  lcd2.clear();
}
